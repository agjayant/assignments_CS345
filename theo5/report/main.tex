\section{Dynamic Reachability}

\begin{algorithm}
%\caption{R}
\label{dyn}
\begin{algorithmic}[1]
\Procedure{Update-R$(i,j)$}{}
\If { R[i] is true and R[j] is false }
\State R[j] $\gets$ true
\For { each neighbor q of j}
\State R[q] $\gets$ true 
\EndFor
\EndIf
\EndProcedure
\end{algorithmic}
\end{algorithm}
